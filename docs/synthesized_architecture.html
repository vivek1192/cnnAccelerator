// booth_mult.v — key sections

// 1. Booth partial products (generate loop)
genvar i;
generate
  for (i = 0; i < 8; i = i + 1) begin : gen_pp
    wire [1:0] bsel = {ext_b[i+1], ext_b[i]};
    assign pp[i] =
      (bsel == 2'b01) ?  (a_ext <<< i) :
      (bsel == 2'b10) ? -(a_ext <<< i) :
      20'sd0;
  end
endgenerate

// 2. Sum of partial products (→ DSP P output)
wire signed [19:0] booth_product;
assign booth_product =
  pp[0]+pp[1]+pp[2]+pp[3]
 +pp[4]+pp[5]+pp[6]+pp[7];

// 3. Correction for unsigned multiplier
wire signed [19:0] correction =
  b[7] ? (a_ext <<< 8) : 20'sd0;

// Vivado inference: recognizes partial product
// summation tree as multiply → infers DSP48E2.
// The correction mux maps to the DSP pre-adder

Vivado Flatten / Optimize Note
Vivado 2023.1 with -flatten_hierarchy rebuilt will merge booth_mult into conv_layer and optimize the partial product tree into 9 DSP48E2 cascade chains. The correction term (b[7] branch) may be fully absorbed into the DSP48E2 pre-adder (D input path), eliminating the surrounding LUTs and reducing the total LUT count from ~184 to approximately ~112.
With -flatten_hierarchy full, Vivado may additionally recognize that weights w[1], w[4], w[7] = 0 (zero multiplications) and eliminate DSP#1, DSP#4, DSP#7 entirely, reducing DSP count from 9 to 6 DSP48E2 for the Sobel kernel specifically.

relu.v — Combinational Block

always @(in) begin
  if (in[15])
    out = 8'd0;
  else if (|in[15:8])
    out = 8'd255;
  else
    out = in[7:0];
end
// → ~2 LUT6, no FFs

max_pool.v — Registered Counter

reg [1:0] count_r;
if (count_r == 2'd3) begin
  pooled_out <= max_result;
  out_valid  <= 1'b1;
  max_val    <= 8'd0;
  count_r    <= 2'd0;
end else begin
  out_valid <= 1'b0;
  count_r   <= count_r + 2'd1;
end
// → 2 FDRE + 1 LUT6

conv_layer.v — Saturator

wire signed [15:0] sat_val;
assign sat_val =
  (sum > 20'sh07FFF) ? 16'sh7FFF :
  (sum < 20'shF8000) ? 16'sh8000 :
  sum[15:0];
// → 2× 20-bit comparators
//   + 2-way mux (LUT chain)
